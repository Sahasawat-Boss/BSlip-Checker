import { NextRequest, NextResponse } from "next/server";

// ✅ Handle GET request (for testing in the browser)
export async function GET() {
    return NextResponse.json({ message: "API is working!" });
}

// ✅ Handle POST request (for API functionality)
export async function POST(request: NextRequest) {
    try {
        const formData = await request.formData();

        const apiKey = process.env.API_EasySlip_KEY;
        if (!apiKey) {
            return NextResponse.json({ error: "Missing API key" }, { status: 500 });
        }

        // Log the API key for debugging (ensure it is properly retrieved)
        console.log("Using API Key:", apiKey);

        // Send the form data with the API key to EasySlip API
        const res = await fetch("https://developer.easyslip.com/api/v1/verify", {
            method: "POST",
            headers: {
                Authorization: `Bearer ${apiKey}`,
            },
            body: formData,
        });

        const data = await res.json();

        if (!res.ok) {
            // Log the response for further debugging
            console.error("Error response from EasySlip API:", data);
            return NextResponse.json({ error: data.message || "Failed to verify data", details: data }, { status: res.status });
        }

        return NextResponse.json({ data });

    } catch (error) {
        console.error("Error while fetching data:", error); // Log the error
        return NextResponse.json({ error: "Failed to verify data", details: String(error) }, { status: 500 });
    }
}
